class Button
BEGINCLASS
	function isMouseOver()
	BEGINFUNCTION
		IF mouse is over this THEN
			return True
		ELSE
			return False
		ENDIF
	ENDFUNCTION

	function render()
	BEGINFUNCTION
		SET stroke to this.outline
		IF isMouseOver() THEN
			SET fill to this.hoverColour
		ELSE
			SET fill to this.baseColour
		ENDIF
		DRAW rectangle
		SET textSize to this.my_textSize
		DRAW text
	ENDFUNCTION
ENDCLASS
